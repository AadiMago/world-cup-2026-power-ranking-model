CREATE TABLE wc_matches AS
SELECT *
FROM last10_matches_filtered
WHERE country IN (
'Mexico','South Africa','South Korea','Czech Republic',
'Canada','Bosnia & Herzegovina','Qatar','Switzerland',
'Brazil','Morocco','Haiti','Scotland',
'United States','Paraguay','Australia','Turkey',
'Germany','Curaçao','Ivory Coast','Ecuador',
'Netherlands','Japan','Sweden','Tunisia',
'Belgium','Egypt','Iran','New Zealand',
'Spain','Cape Verde','Saudi Arabia','Uruguay',
'France','Senegal','Iraq','Norway',
'Argentina','Algeria','Austria','Jordan',
'Portugal','DR Congo','Uzbekistan','Colombia',
'England','Croatia','Ghana','Panama'
);

UPDATE wc_matches
SET win = CASE
    WHEN goals_scored > goals_conceded THEN 3
    WHEN goals_scored = goals_conceded THEN 1
    ELSE 0
END;

CREATE TABLE average_stats AS
SELECT
    country,
    COUNT(*) AS games_played,
    AVG(goals_scored) AS avg_goals_scored,
    AVG(goals_conceded) AS avg_goals_conceded,
    AVG(goals_scored - goals_conceded) AS avg_goal_diff,
    AVG(win) AS points_per_game
FROM wc_matches
GROUP BY country;

ALTER TABLE average_stats ADD COLUMN world_cup_appearances INTEGER;
ALTER TABLE average_stats ADD COLUMN world_cup_medals INTEGER;

ALTER TABLE average_stats ADD COLUMN continental_titles_10y INTEGER;

SET continental_titles_10y = CASE country
    WHEN 'Argentina' THEN 2
    WHEN 'Brazil' THEN 1
    WHEN 'France' THEN 1
    WHEN 'Germany' THEN 1
    WHEN 'Portugal' THEN 3

ALTER TABLE average_stats ADD COLUMN power_score REAL;

UPDATE average_stats
SET power_score =
    0.10 * avg_goal_diff +
    0.35 * (points_per_game / 3.0) +
    0.35 * ((100 - fifa_rank)/100.0) * ((100 - fifa_rank)/100.0) +
    0.05 * world_cup_medals +
    0.05 * world_cup_appearances +
    0.10 * continental_titles_10y;

ALTER TABLE average_stats ADD COLUMN contender_status TEXT;

UPDATE average_stats
SET team_tier = CASE 
    WHEN power_score >= 2 THEN 'Favourite'
    WHEN power_score >= 1.4 THEN 'Contender'
    WHEN power_score >= 0.8 THEN 'Dark Horse'
    WHEN power_score >= 0.4 THEN 'Unlikely'
    ELSE 'Highly Improbable'
END;
